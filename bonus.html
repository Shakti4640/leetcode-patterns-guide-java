FIRST THING TO TRY

// FIXED SIZE WINDOW:

for (int i = 0; i < k; i++) windowSum += arr[i];  // first window
int maxSum = windowSum;
for (int i = k; i < arr.length; i++) {
    windowSum += arr[i] - arr[i - k];              // slide
    maxSum = Math.max(maxSum, windowSum);
}

// VARIABLE SIZE WINDOW:

        // SHRINK: remove arr[windowStart] from window state
        windowStart++;
    }
    // UPDATE: result = Math.max(result, windowEnd - windowStart + 1);
}

FIRST THING TO TRY

// CYCLE DETECTION:
ListNode slow = head, fast = head;
while (fast != null && fast.next != null) {
    slow = slow.next;
    fast = fast.next.next;
    if (slow == fast) return true;  // cycle exists
}
return false;  // no cycle

// FIND MIDDLE:
ListNode slow = head, fast = head;
while (fast != null && fast.next != null) {
    slow = slow.next;
    fast = fast.next.next;
}
return slow;  // slow is at middle

// FIND CYCLE START:
// Phase 1: detect meeting point (above)
// Phase 2: move one pointer to head, advance both by 1
ListNode pointer1 = head;
while (pointer1 != slow) {
    pointer1 = pointer1.next;
    slow = slow.next;
}
return pointer1;  // cycle start

BEST COMPLEXITY ACHIEVABLE

Time:  O(n)        — fast pointer traverses at most 2n steps
Space: O(1)        — only two pointer variables

// Add each node; if add() returns false → cycle found
// But this is O(n) space — Fast/Slow is O(1)

// For palindrome check via fast/slow:
// Find middle → reverse second half → compare → restore
// No collections needed

DATA SHAPE GIVEAWAYS

→ Numbers are in a KNOWN RANGE [0..n] or [1..n]
→ Array length is CLOSE to the range of values
→ Each number SHOULD map to a specific index
→ Problem says "without extra space" or "O(1) space"
→ There is a natural INDEX ↔ VALUE mapping

FIRST THING TO TRY

// CYCLIC SORT (place each number at its correct index):
int i = 0;
while (i < nums.length) {
    int correctIndex = nums[i] - 1;  // for range [1, n]
    // int correctIndex = nums[i];    // for range [0, n]
    if (nums[i] > 0 && nums[i] <= nums.length && nums[i] != nums[correctIndex]) {
        // swap nums[i] and nums[correctIndex]
        int temp = nums[i];
        nums[i] = nums[correctIndex];
        nums[correctIndex] = temp;
    } else {
        i++;
    }
}
// After sort: index i should hold value i+1 (for [1,n])
// Any mismatch reveals missing/duplicate numbers

BEST COMPLEXITY ACHIEVABLE

Time:  O(n)        — each number is swapped at most once to its correct position
Space: O(1)        — in-place swaps only

JAVA COLLECTIONS THAT HELP

for (int j = 0; j <= n; j++) xor ^= j;

// xor now holds the missing number

  ├── YES → Find ALL duplicates? → CYCLIC SORT + negative marking
  ├── YES → Smallest missing positive? → CYCLIC SORT variant

// LEVEL ORDER TRAVERSAL:
List<List<Integer>> result = new ArrayList<>();
if (root == null) return result;

Queue<TreeNode> queue = new LinkedList<>();
queue.offer(root);

while (!queue.isEmpty()) {
    int levelSize = queue.size();           // KEY: snapshot current level size
    List<Integer> currentLevel = new ArrayList<>();

        TreeNode node = queue.poll();
        currentLevel.add(node.val);
        if (node.left != null) queue.offer(node.left);

    result.add(currentLevel);
}

// PATH SUM (does any root-to-leaf path sum to target?):
public boolean hasPathSum(TreeNode root, int targetSum) {
    if (root == null) return false;
    if (root.left == null && root.right == null) return targetSum == root.val;  // leaf
    return hasPathSum(root.left, targetSum - root.val) 
        || hasPathSum(root.right, targetSum - root.val);
}

// ALL PATHS WITH TARGET SUM (backtracking):
public void findPaths(TreeNode node, int remaining, List<Integer> path, 
                       List<List<Integer>> result) {
    if (node == null) return;
    path.add(node.val);
    
    if (node.left == null && node.right == null && remaining == node.val) {
        result.add(new ArrayList<>(path));   // KEY: copy the path
    } else {
        findPaths(node.left, remaining - node.val, path, result);
        findPaths(node.right, remaining - node.val, path, result);
    }
    path.remove(path.size() - 1);           // BACKTRACK
}

BEST COMPLEXITY ACHIEVABLE

Time:  O(n)        — visit every node once (for single-answer problems)
       O(n²)       — for all-paths problems (n paths of length n each)
Space: O(h)        — recursion stack depth = height of tree
       O(n)        — worst case for skewed tree

FIRST THING TO TRY

// MEDIAN FROM DATA STREAM:
PriorityQueue<Integer> maxHeap = new PriorityQueue<>(Collections.reverseOrder()); // lower half
PriorityQueue<Integer> minHeap = new PriorityQueue<>();                           // upper half

public void addNum(int num) {
    maxHeap.offer(num);                          // add to lower half
    minHeap.offer(maxHeap.poll());               // balance: move max of lower to upper
    if (minHeap.size() > maxHeap.size()) {       // keep maxHeap same size or 1 bigger
        maxHeap.offer(minHeap.poll());
    }
}

public double findMedian() {
    if (maxHeap.size() > minHeap.size()) return maxHeap.peek();
    return (maxHeap.peek() + minHeap.peek()) / 2.0;
}

BEST COMPLEXITY ACHIEVABLE

Time:  O(log n) per insertion, O(1) per median query
Space: O(n)     — storing all elements across both heaps

// ALL SUBSETS:
List<List<Integer>> result = new ArrayList<>();
public void backtrack(int[] nums, int start, List<Integer> current) {
    result.add(new ArrayList<>(current));          // every state is a valid subset
    for (int i = start; i < nums.length; i++) {
        current.add(nums[i]);                      // choose
        backtrack(nums, i + 1, current);           // explore
        current.remove(current.size() - 1);        // un-choose (backtrack)
    }
}

// ALL PERMUTATIONS:
public void permute(int[] nums, List<Integer> current, boolean[] used) {
    if (current.size() == nums.length) {
        result.add(new ArrayList<>(current));
        return;
    }
    for (int i = 0; i < nums.length; i++) {
        if (used[i]) continue;
        used[i] = true;
        current.add(nums[i]);
        permute(nums, current, used);
        current.remove(current.size() - 1);
        used[i] = false;
    }
}

// COMBINATION SUM (elements reusable):
public void combo(int[] candidates, int target, int start, List<Integer> current) {
    if (target == 0) { result.add(new ArrayList<>(current)); return; }
    if (target < 0) return;  // pruning
    for (int i = start; i < candidates.length; i++) {
        current.add(candidates[i]);
        combo(candidates, target - candidates[i], i, current);  // i not i+1 (reuse allowed)
        current.remove(current.size() - 1);
    }
}

// KAHN'S ALGORITHM (BFS-based topological sort):
// Step 1: Build adjacency list and in-degree array
Map<Integer, List<Integer>> graph = new HashMap<>();
int[] inDegree = new int[numNodes];

for (int[] edge : edges) {
    graph.computeIfAbsent(edge[0], k -> new ArrayList<>()).add(edge[1]);
    inDegree[edge[1]]++;
}

// Step 2: Start with all nodes having in-degree 0
Queue<Integer> queue = new LinkedList<>();
for (int i = 0; i < numNodes; i++) {
    if (inDegree[i] == 0) queue.offer(i);
}

// Step 3: Process BFS — remove node, reduce neighbors' in-degree
List<Integer> topoOrder = new ArrayList<>();
while (!queue.isEmpty()) {
    int node = queue.poll();
    topoOrder.add(node);
    for (int neighbor : graph.getOrDefault(node, Collections.emptyList())) {
        inDegree[neighbor]--;
        if (inDegree[neighbor] == 0) queue.offer(neighbor);
    }
}

// Step 4: Check if all nodes were processed (no cycle)
if (topoOrder.size() != numNodes) {
    // CYCLE EXISTS → topological sort impossible
    return new int[0];
}
return topoOrder.stream().mapToInt(i -> i).toArray();

// UNION FIND with Path Compression + Union by Rank:
class UnionFind {
    int[] parent;
    int[] rank;
    int components;
    
    UnionFind(int n) {
        parent = new int[n];
        rank = new int[n];
        components = n;
        for (int i = 0; i < n; i++) parent[i] = i;
    }
    
    int find(int x) {
        if (parent[x] != x) {
            parent[x] = find(parent[x]);   // PATH COMPRESSION
        }
        return parent[x];
    }
    
    boolean union(int x, int y) {
        int rootX = find(x), rootY = find(y);
        if (rootX == rootY) return false;   // already connected
        
        // UNION BY RANK
        if (rank[rootX] < rank[rootY]) parent[rootX] = rootY;
        else if (rank[rootX] > rank[rootY]) parent[rootY] = rootX;
        else { parent[rootY] = rootX; rank[rootX]++; }
        
        components--;
        return true;                        // successfully merged
    }
    
    boolean connected(int x, int y) {
        return find(x) == find(y);
    }
}

// TRIE OPERATIONS:
class Trie {
    TrieNode root = new TrieNode();
    
    // INSERT: O(L) where L = word length
    void insert(String word) {
        TrieNode curr = root;
        for (char ch : word.toCharArray()) {
            int idx = ch - 'a';
            if (curr.children[idx] == null) {
                curr.children[idx] = new TrieNode();
            }
            curr = curr.children[idx];
        }
        curr.isEndOfWord = true;
    }
    
    // SEARCH: O(L)
    boolean search(String word) {
        TrieNode node = findNode(word);
        return node != null && node.isEndOfWord;
    }
    
    // STARTS WITH (prefix check): O(L)
    boolean startsWith(String prefix) {
        return findNode(prefix) != null;
    }
    
    private TrieNode findNode(String s) {
        TrieNode curr = root;
        for (char ch : s.toCharArray()) {
            int idx = ch - 'a';
            if (curr.children[idx] == null) return null;
            curr = curr.children[idx];
        }
        return curr;
    }
}
